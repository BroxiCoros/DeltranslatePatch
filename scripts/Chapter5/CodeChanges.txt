{

    {
        var breaks = 0;
        var max_len = 0;
        var cur_len = 0;

        for (var j = 1; j <= string_length(choicetext[i]); j++)
        {
            if (string_char_at(choicetext[i], j) == "\n")
            {
                max_len = max(max_len, cur_len);
                cur_len = 0;
                breaks++;
            }
            else
            {
                cur_len++;
            }

            max_len = max(max_len, cur_len);
        }

        switch (breaks)
        {
            case 0:
                choicetextxscale[i] = max(1, 3 - floor(max(0, max_len - 4) / 4));
                choicetextyscale[i] = max(1, 3 - (max(0, max_len - 7) * 0.2));
                break;

            case 1:
                choicetextxscale[i] = max(1, round(3 - floor(max(0, max_len - 4) / 4)));
                choicetextyscale[i] = 76 / string_height(choicetext[i]);
                break;

            default:
                choicetextxscale[i] = max(1, round(3 - floor(max(0, max_len - 4) / 4)));
                choicetextyscale[i] = round(62 / string_height(choicetext[i]));
                break;
        }
    }
}
